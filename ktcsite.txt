  <script>
    // ===== 기능(등록/조회/삭제/검증)은 그대로 유지 =====

    // ===== 데이터 =====
    const uid = () => (crypto?.randomUUID ? crypto.randomUUID() : String(Date.now()) + Math.random().toString(16).slice(2));
    let products = [
      { id: uid(), name:"프리미엄 멤버십 1개월", category:"구독/멤버십", price:12000, status:"판매중", desc:"전용 콘텐츠 + 월 1회 Q&A", createdAt:Date.now()-20000 },
      { id: uid(), name:"1:1 컨설팅 (60분)", category:"컨설팅", price:150000, status:"중지", desc:"요구사항 정리 및 개선 포인트 피드백", createdAt:Date.now()-10000 }
    ];

    // ===== DOM =====
    const form = document.getElementById('productForm');
    const listBody = document.getElementById('listBody');
    const countEl = document.getElementById('count');
    const noticeEl = document.getElementById('notice');

    const nameEl = document.getElementById('name');
    const categoryEl = document.getElementById('category');
    const priceEl = document.getElementById('price');
    const descEl = document.getElementById('desc');

    // ===== 유틸 =====
    const escapeHtml = s => String(s).replace(/[&<>"']/g,m=>({ '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#039;' }[m]));
    const moneyKRW = n => new Intl.NumberFormat('ko-KR').format(n)+'원';
    const badge = s => `<span class="badge ${s==='판매중'?'on':'off'}">${escapeHtml(s)}</span>`;

    function clampDescTo2Lines(text){
      const lines = String(text).replace(/\r\n/g, '\n').split('\n');
      return lines.slice(0,2).join('\n');
    }

    function showNotice(message, type){
      noticeEl.textContent = message || '';
      noticeEl.classList.toggle('show', !!message);
      noticeEl.classList.toggle('error', type === 'error');
    }

    // ===== 렌더 =====
    function render(){
      countEl.textContent = products.length;
      listBody.innerHTML = products
        .slice()
        .sort((a,b)=>b.createdAt-a.createdAt)
        .map(p=>`
          <tr data-id="${p.id}">
            <td>
              <div class="product-cell">
                <div class="thumb" aria-hidden="true">
                  <img src="card-thumb.jpg" alt="" />
                </div>
                <div>
                  <p class="name">${escapeHtml(p.name)}</p>
                  ${p.desc ? `<div class="desc">${escapeHtml(p.desc)}</div>` : ``}
                </div>
              </div>
            </td>
            <td>${escapeHtml(p.category)}</td>
            <td class="price">${moneyKRW(p.price)}</td>
            <td>${badge(p.status)}</td>
            <td><button type="button" class="btn-del" data-action="delete" data-id="${p.id}">삭제</button></td>
          </tr>
        `).join('');
    }

    // ===== 가격 입력: 숫자만 최대한 제한(기능 유지) =====
    priceEl.addEventListener('keydown', (e) => {
      const allowedKeys = [
        'Backspace','Delete','Tab','Enter','Escape',
        'ArrowLeft','ArrowRight','ArrowUp','ArrowDown','Home','End'
      ];
      const ctrlCmd = e.ctrlKey || e.metaKey;
      if (allowedKeys.includes(e.key)) return;
      if (ctrlCmd && ['a','c','v','x'].includes(e.key.toLowerCase())) return;

      if (!/^\d$/.test(e.key)){
        e.preventDefault();
      }
    });

    priceEl.addEventListener('input', () => {
      const onlyDigits = priceEl.value.replace(/[^\d]/g, '');
      if (priceEl.value !== onlyDigits) priceEl.value = onlyDigits;
    });

    // ===== 등록(검증 포함, 기능 유지) =====
    form.addEventListener('submit', (e) => {
      e.preventDefault();

      const name = nameEl.value.trim();
      const category = categoryEl.value;
      const priceStr = String(priceEl.value || '').trim();
      const status = form.elements['status']?.value;

      if (!name || !category || !priceStr || !status){
        showNotice('필수 항목(상품명/카테고리/가격/상태)을 입력하세요.', 'error');
        return;
      }

      const price = Number(priceStr);
      if (!Number.isFinite(price) || priceStr === '' || price < 0){
        showNotice('가격은 0 이상의 숫자만 입력할 수 있어요.', 'error');
        return;
      }

      const desc = clampDescTo2Lines(descEl.value.trim());

      products.push({
        id: uid(),
        name,
        category,
        price: Math.floor(price),
        status,
        desc,
        createdAt: Date.now()
      });

      render();

      form.reset();
      form.elements['status'].value = '판매중';
      descEl.value = '';
      showNotice('등록되었습니다.', 'ok');
      nameEl.focus();
    });

    // ===== 삭제(기능 유지) =====
    listBody.addEventListener('click', (e) => {
      const btn = e.target.closest('[data-action="delete"]');
      if (!btn) return;

      const id = btn.dataset.id;
      products = products.filter(p => p.id !== id);
      render();
      showNotice('삭제되었습니다.', 'ok');
    });

    render();
  </script>
